265auto obj::read(const std::string& file_name) -> void {
266 std::ifstream file(file_name);
267 if (!file.is_open()) throw std::runtime_error("[ERR] Could not open file");
268
269 std::string buf{ std::istreambuf_iterator<char>(file), std::istreambuf_iterator<char>() };
270
271 const char* p = buf.data();
272 const char* end = p + buf.size();
273
274 auto skip_ws = [](const char* p, const char* end) noexcept -> const char* {
275 while (p < end && (*p == ' ' || *p == '\t')) ++p;
276 return p;
277 };
278
279 auto skip_line = [](const char* p, const char* end) noexcept -> const char* {
280 while (p < end && *p != '\n') ++p;
281 return p < end ? p + 1 : p;
282 };

284 auto read_float = [&](const char* p, float& out) noexcept -> const char* {
285 p = skip_ws(p, end);
286 auto [ptr, _] = std::from_chars(p, end, out);
287 return ptr;
288 };

290 auto read_face_token = [&](const char* p, _face_vertex& fv) noexcept -> const char* {
291 p = skip_ws(p, end);
292 auto [p1, _1] = std::from_chars(p, end, fv.v); p = p1;
293 if (p < end && *p == '/') {
294 ++p;
295 if (p < end && *p != '/') { auto [p2, _2] = std::from_chars(p, end, fv.vt); p = p2; }
296 if (p < end && *p == '/') { ++p; auto [p3, _3] = std::from_chars(p, end, fv.vn); p = p3; }
297 }
298 return p;
299 };
300
301 while (p < end) {
302 p = skip_ws(p, end);
303 if (p >= end) break;
304 if (*p == '#' || *p == '\n' || *p == '\r') { p = skip_line(p, end); continue; }
305
306 const char* tok = p;
307 while (p < end && *p != ' ' && *p != '\t' && *p != '\n' && *p != '\r') ++p;
308 const std::string_view kw{ tok, static_cast<std::size_t>(p - tok) };
309
310 if (kw == "v") {
311 _vertex v{};
312 p = read_float(p, v.x);
313 p = read_float(p, v.y);
314 p = read_float(p, v.z);
315 p = read_float(p, v.w);
316 v_.push_back(v);
317 } else if (kw == "vt") {
318 _texture_coordinate vt{};
319 p = read_float(p, vt.u);
320 p = read_float(p, vt.v);
321 p = read_float(p, vt.w);
322 vt_.push_back(vt);
323 } else if (kw == "vn") {
324 _vertex_normal vn{};
325 p = read_float(p, vn.x);
326 p = read_float(p, vn.y);
327 p = read_float(p, vn.z);
328 vn_.push_back(vn);
329 } else if (kw == "vp") {
330 _parameter_space_vertex vp{};
331 p = read_float(p, vp.u);
332 p = read_float(p, vp.v);
333 p = read_float(p, vp.w);
334 vp_.push_back(vp);
335 } else if (kw == "f") {
336 _face f{};
337 p = skip_ws(p, end);
338 while (p < end && *p != '\n' && *p != '\r') {
339 _face_vertex fv{};
340 p = read_face_token(p, fv);
341 f.push_back(fv);
342 p = skip_ws(p, end);
343 }
344 f_.push_back(std::move(f));
345 } else if (kw == "l") {
346 _line l{};
347 p = skip_ws(p, end);
348 while (p < end && *p != '\n' && *p != '\r') {
349 _line_vertex lv{};
350 auto [ptr, _] = std::from_chars(p, end, lv);
351 p = ptr;
352 l.push_back(lv);
353 p = skip_ws(p, end);
354 }
355 l_.push_back(std::move(l));
356 }
357
358 p = skip_line(p, end);
359 }
