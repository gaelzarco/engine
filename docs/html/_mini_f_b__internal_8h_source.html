12// Mobile backends may encode a pointer id in the upper bits of mouse positions.
13// Keep these constants centralized so packing/unpacking stays compatible.
14//-------------------------------------
15#define MFB_COMBINED_POS_ID_BITS 4u
16#define MFB_COMBINED_POS_BITS (32u - MFB_COMBINED_POS_ID_BITS)
17#define MFB_COMBINED_POS_ID_SHIFT MFB_COMBINED_POS_BITS
18#define MFB_COMBINED_POS_MASK ((1u << MFB_COMBINED_POS_BITS) - 1u)
19#define MFB_COMBINED_POS_SIGN_BIT (1u << (MFB_COMBINED_POS_BITS - 1u))
20#define MFB_COMBINED_ID_MASK ((1u << MFB_COMBINED_POS_ID_BITS) - 1u)
21
22//-------------------------------------
23static inline int
24mfb_pack_pos_id(int32_t pos, uint32_t id) {
25 uint32_t packed = (((uint32_t) pos) & MFB_COMBINED_POS_MASK)
26 | ((id & MFB_COMBINED_ID_MASK) << MFB_COMBINED_POS_ID_SHIFT);
27 return (int) packed;
28}

30//-------------------------------------
31static inline int32_t
32mfb_unpack_pos(uint32_t combined) {
33 uint32_t pos_bits = combined & MFB_COMBINED_POS_MASK;
34 if ((pos_bits & MFB_COMBINED_POS_SIGN_BIT) != 0u) {
35 pos_bits |= ~MFB_COMBINED_POS_MASK;
36 }
37 return (int32_t) pos_bits;
38}
